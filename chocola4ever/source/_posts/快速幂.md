#include<bits/stdc++.h>
#define ll long long
using namespace std;
inline ll qpow(ll a,ll b)
{
    ll ans=1;
    while (b)
    {
        if (b%2==1)
            ans*=a;
        a*=a;
        b>>=1;
    }
    return ans;
}
int main(){
    cout<<qpow(0,0)<<endl;
}
