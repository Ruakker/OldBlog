## 一行gcd

```cpp
inline ll gcd(ll a,ll b){return !b?a:gcd(b,a%b);}
```

## 快速幂

```cpp
#define ll long long
inline ll qpow(int a,int b){
	ll ans=1;
	while (b){
		if (b&1) ans*=a;
		a*=a,b>>=1;
	}
	return ans;
}
```

## 分数取模 $\frac{a}{b} \% Mod=a \times b^{Mod-2} \% Mod$

```cpp
#define ll long long
const int Mod=998244353;
inline ll qpow(ll a,ll b){
    ll ans=1;
    while (b){
        if (b&1)
            ans=a*ans%Mod;
        a=a*a%Mod;
        b>>=1;
    }
    return ans;
}
int main(){
	//	...
	printf("%d",a*qpow(b,Mod-2)%Mod);
	//	...
}
```
